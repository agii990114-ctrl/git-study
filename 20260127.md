# fastapi 파라미터

``` py
from fastapi import FastAPI
from fastapi.responses import HTMLResponse

app = FastAPI() #was 서버를 켜기 위한 선언 // 변수에 담아줘야하며 시작하는 부분에 선언

arr = []


@app.get("/", response_class=HTMLResponse)
def root(txt:str =""):
   
    print(f"전역 배열 : {arr}")
    print(f"전달 받은 변수 : {txt}")
    if txt == "":
        return """
        <body>
            <form>
                <input type="text" name="txt" />
                <button type="submit">전송</button>
            </form>
        </body>
    """
    else:
        arr.append(txt)
        html = ""
        for v in arr:
            html += f"<li>{v}</li>"

        return f"""
        <body>
            <ul>
                {html}
            </ul>
            <a href="/">돌아가기</a>
        </body>
    """

```
