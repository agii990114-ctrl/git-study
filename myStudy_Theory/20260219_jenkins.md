## dev-ops
1. dev : 개발
2. ops : 운영
3. ci / cd : dev-ops를 제어
4. hudson / jenkins


```
pipeline {
	agent any
	environment {
	    IMAGE_NAME = "backend"
	    CONTAINER_NAME = "fastapi"
		GITHUB_REPO = "https://github.com/agii990114-ctrl/Jenkins_Test.git"
	}
	stages {
		stage('Checkout') {
			steps {
				echo "Checking out code from GitHub..."
				git branch: 'main', url: "${GITHUB_REPO}"
			}
		}
		stage('Build Docker Image') {
			steps {
			    sh 'docker build -t $IMAGE_NAME .'
			}
		}
		stage('Run Container') {
			steps {
			    sh '''
			        docker rm -f $CONTAINER_NAME || true
			        docker run -i -d -p 8000:8000 --name $CONTAINER_NAME $IMAGE_NAME
                '''
			}
		}
	}
	post {
		success {
			echo 'Checkout completed successfully!'
		}
		failure {
			echo 'Checkout failed.'
		}
		always {
		    echo 'End'
		}
	}
}
```
